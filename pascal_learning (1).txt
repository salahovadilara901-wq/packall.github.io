let fcIndex = 0;
let fcOrder = [];
let fcFlipped = false;

function initFlashcards() {
  fcOrder = [...Array(flashcardsData.length).keys()];
  fcIndex = 0;
  fcFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
  updateFlashcard();
}

function updateFlashcard() {
  const card = flashcardsData[fcOrder[fcIndex]];
  document.getElementById('fc-question').textContent = card.q;
  document.getElementById('fc-answer').textContent = card.a;
  document.getElementById('fc-counter').textContent = `Карточка ${fcIndex + 1} из ${flashcardsData.length}`;
  document.getElementById('fc-progress').style.width = `${((fcIndex + 1) / flashcardsData.length) * 100}%`;
  fcFlipped = false;
  document.getElementById('flashcard').classList.remove('flipped');
}

function flipCard() {
  fcFlipped = !fcFlipped;
  document.getElementById('flashcard').classList.toggle('flipped', fcFlipped);
}

function nextCard() {
  fcIndex = (fcIndex + 1) % flashcardsData.length;
  updateFlashcard();
}

function prevCard() {
  fcIndex = (fcIndex - 1 + flashcardsData.length) % flashcardsData.length;
  updateFlashcard();
}

function shuffleCards() {
  for (let i = fcOrder.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [fcOrder[i], fcOrder[j]] = [fcOrder[j], fcOrder[i]];
  }
  fcIndex = 0;
  updateFlashcard();
}

let qIndex = 0;
let qScore = 0;
let qShuffled = [];
let qAnswered = false;

function initQuiz() {
  qShuffled = [...quizData].sort(() => Math.random() - 0.5);
  qIndex = 0;
  qScore = 0;
  qAnswered = false;
  document.getElementById('quiz-result').classList.remove('show');
  document.getElementById('quiz-body').style.display = 'block';
  renderQuestion();
}

function renderQuestion() {
  const q = qShuffled[qIndex];
  document.getElementById('q-num').textContent = `Вопрос ${qIndex + 1} из ${qShuffled.length}`;
  document.getElementById('q-text').textContent = q.q;
  document.getElementById('quiz-progress').style.width = `${(qIndex / qShuffled.length) * 100}%`;

  const letters = ['А', 'Б', 'В', 'Г'];
  const optsEl = document.getElementById('q-options');
  optsEl.innerHTML = '';
  q.opts.forEach((opt, i) => {
    const btn = document.createElement('button');
    btn.className = 'quiz-option';
    btn.innerHTML = `<span class="quiz-option-letter">${letters[i]}</span>${opt}`;
    btn.onclick = () => selectAnswer(i);
    optsEl.appendChild(btn);
  });

  const fb = document.getElementById('q-feedback');
  fb.className = 'quiz-feedback';
  fb.textContent = '';
  document.getElementById('q-next').className = 'quiz-next';
  qAnswered = false;
}

function selectAnswer(i) {
  if (qAnswered) return;
  qAnswered = true;
  const q = qShuffled[qIndex];
  const opts = document.querySelectorAll('.quiz-option');
  opts.forEach(o => o.disabled = true);

  const fb = document.getElementById('q-feedback');
  if (i === q.correct) {
    qScore++;
    opts[i].classList.add('correct');
    fb.textContent = '✓ Верно! ' + q.explain;
    fb.className = 'quiz-feedback correct-fb show';
  } else {
    opts[i].classList.add('wrong');
    opts[q.correct].classList.add('correct');
    fb.textContent = '✗ Неверно. ' + q.explain;
    fb.className = 'quiz-feedback wrong-fb show';
  }

  const nextBtn = document.getElementById('q-next');
  nextBtn.textContent = qIndex < qShuffled.length - 1 ? 'Следующий вопрос →' : 'Посмотреть результат →';
  nextBtn.className = 'quiz-next show';
}

function nextQuestion() {
  qIndex++;
  if (qIndex >= qShuffled.length) {
    showResult();
  } else {
    renderQuestion();
  }
}

function showResult() {
  document.getElementById('quiz-body').style.display = 'none';
  document.getElementById('quiz-progress').style.width = '100%';
  const pct = Math.round((qScore / qShuffled.length) * 100);
  document.getElementById('result-score').textContent = `${qScore} / ${qShuffled.length}`;
  const texts = [
    [0, 40, "Паскаль сказал бы: «Надо работать над собой». Попробуйте ещё раз!"],
    [40, 70, "Неплохое начало! Повторите карточки и попробуйте снова."],
    [70, 90, "Хороший результат! Вы хорошо знаете биографию Паскаля."],
    [90, 101, "Блестяще! Вы знаете Паскаля не хуже, чем он сам себя."]
  ];
  const t = texts.find(([lo, hi]) => pct >= lo && pct < hi);
  document.getElementById('result-text').textContent = t[2];
  document.getElementById('quiz-result').classList.add('show');
}

function restartQuiz() {
  initQuiz();
}
